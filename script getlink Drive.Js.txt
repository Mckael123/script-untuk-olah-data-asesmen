function MencariID() {
var ss=SpreadsheetApp.getActiveSpreadsheet();
var s=ss.getActiveSheet();
var c=s.getRange("x2");
var fldr=DriveApp.getFolderById("1MieE3XPlwxabsSL7lS7giXVTZjjzbQN4");
var files=fldr.getFiles();
var names=[],f,str; 
while (files.hasNext()) {
f=files.next();
str = f.getId();
names.push([str]);
}
s.getRange(c.getRow(),c.getColumn(),names.length).setFormulas(names); }
